## 1) Куди переносимо CanvasArea.js у новій структурі

**Було:**

* `app/components/Scheme/CanvasArea.js`

**Стає (логічно під твою структуру):**

## 2) Оновлений код `src/features/scheme-editor/ui/page/CanvasArea.js`

```jsx
import React, { useState } from "react";

import KonvaStage from "../canvas-area/KonvaStage";
import CanvasHeader from "../canvas-area/CanvasHeader";
import SchemeTable from "../canvas-area/SchemeTable";
import NotesBlock from "../canvas-area/NotesBlock";
import AddTableModal from "../canvas-area/modals/AddTableModal";

const CanvasArea = () => {
  const [elements, setElements] = useState([]);
  const [rows, setRows] = useState([]);
  const [note, setNote] = useState("");
  const [isTableModalOpen, setTableModalOpen] = useState(false);
  const [noteOffset, setNoteOffset] = useState(0);

  const handleAddRow = (row) => {
    setRows([...rows, row]);
  };

  return (
    <div
      style={{
        position: "relative",
        width: "1123px",
        height: "794px",
        margin: "auto",
        border: "1px solid black",
      }}
    >
      <CanvasHeader
        albumCode="3.ТГ.000-СШ"
        point="55"
        system="Q1"
        schemeName="Схема розміщення системи Q1"
        sheetNumber="3"
      />

      <KonvaStage width={1123} height={794} elements={elements} />

      <SchemeTable
        rows={rows}
        onAddRow={() => setTableModalOpen(true)}
        onNotePositionUpdate={(count) => setNoteOffset(count * 30)}
      />

      <NotesBlock note={note} noteOffset={noteOffset} />

      <AddTableModal
        open={isTableModalOpen}
        onClose={() => setTableModalOpen(false)}
        onAddRow={handleAddRow}
      />
    </div>
  );
};

export default CanvasArea;
```
