## ‚úÖ –í–∏–ø—Ä–∞–≤–ª–µ–Ω–∏–π `TopHeaderLine.js`

```jsx
"use client";

import Link from "next/link";
import { useRouter } from "next/navigation";
import { useAuth } from "@/context/AuthContext";

const PROXY_URL = process.env.NEXT_PUBLIC_PROXY_URL;

export default function TopHeaderLine() {
  const { user } = useAuth();
  const router = useRouter();

  const handleLogout = async () => {
    try {
      await fetch(`${PROXY_URL}/api/rest/logoutrest`, {
        method: "POST",
        credentials: "include",
        headers: {
          "Content-Type": "application/json",
        },
      });
    } catch (error) {
      console.error("Logout error:", error);
    } finally {
      router.push("/login");
    }
  };

  return (
    <div style={navBarStyle}>
      {/* –õ—ñ–≤–∞ —á–∞—Å—Ç–∏–Ω–∞ */}
      <div style={menuStyle}>
        <Link href="/" style={menuItemStyle}>
          Main
        </Link>

        <Link href="/schemes" style={menuItemStyle}>
          Scheme Catalog
        </Link>

        <Link href="/editor" style={menuItemStyle}>
          Editor
        </Link>

        <Link href="/create-scheme" style={menuItemStyle}>
          Add New
        </Link>
      </div>

      {/* –ü—Ä–∞–≤–∞ —á–∞—Å—Ç–∏–Ω–∞ */}
      <div style={userInfoStyle}>
        {user?.login && <span style={userNameStyle}>{user.login}</span>}

        <button onClick={handleLogout} style={logoutButtonStyle}>
          Logout
        </button>
      </div>
    </div>
  );
}

/* ================= styles ================= */

const navBarStyle = {
  position: "fixed",
  top: 0,
  left: 0,
  width: "100%",
  height: "60px",
  backgroundColor: "#333",
  color: "#fff",
  display: "flex",
  alignItems: "center",
  justifyContent: "space-between",
  boxShadow: "0px 4px 6px rgba(0, 0, 0, 0.1)",
  zIndex: 1000,
};

const menuStyle = {
  display: "flex",
  alignItems: "center",
  gap: "20px",
  paddingLeft: "15px",
};

const menuItemStyle = {
  color: "#fff",
  textDecoration: "none",
  fontSize: "16px",
  fontWeight: "bold",
  cursor: "pointer",
};

const userInfoStyle = {
  display: "flex",
  alignItems: "center",
  gap: "15px",
  paddingRight: "15px",
};

const userNameStyle = {
  fontSize: "16px",
  fontWeight: "bold",
};

const logoutButtonStyle = {
  backgroundColor: "red",
  color: "white",
  border: "none",
  padding: "8px 12px",
  fontSize: "14px",
  fontWeight: "bold",
  cursor: "pointer",
  borderRadius: "5px",
};
```
