* без зайвої вкладеності
* без prop drilling
* із **Zustand**
* з **central controller**
* з можливістю далі спокійно розширювати `useDiagramState` логіку, але вже не у вигляді моноліту

# 4. Створюємо Zustand store

Це буде **центральне місце стану**.

## `store/useCanvasStore.js`

```js
import { create } from 'zustand';
import { initialSymbols, initialConnections } from '@/lib/diagram/initialData';

export const useCanvasStore = create((set, get) => ({
  // ===== DATA =====
  symbols: initialSymbols,
  connections: initialConnections,

  // ===== UI =====
  selection: null, // { type: 'node' | 'connection', id }
  hoveredElementId: null,
  hoveredAnchor: null,
  hoveredConnectionId: null,
  hoveredLineAnchorId: null,

  // ===== MODES =====
  mode: null, // 'add-connector' | 'add-anchor' | null
  selectedAnchor: null,
  message: '',

  // ===== CONTEXT MENU =====
  contextMenu: {
    open: false,
    position: null, // { top, left }
    target: null,   // { type, id }
  },

  // ===== STAGE / REFS =====
  stageRef: null,
  layerRef: null,
  apiRef: null,

  setStageRef: (ref) => set({ stageRef: ref }),
  setLayerRef: (ref) => set({ layerRef: ref }),
  setApiRef: (ref) => set({ apiRef: ref }),

  // ===== SYMBOLS =====
  setSymbols: (symbols) => set({ symbols }),

  addSymbol: (symbol) =>
    set((state) => ({
      symbols: [...state.symbols, symbol],
    })),

  updateSymbolPosition: (id, x, y) =>
    set((state) => ({
      symbols: state.symbols.map((item) =>
        item.id === id ? { ...item, x, y } : item
      ),
    })),

  removeSymbol: (id) =>
    set((state) => ({
      symbols: state.symbols.filter((item) => item.id !== id),
    })),

  // ===== CONNECTIONS =====
  setConnections: (connections) => set({ connections }),

  addConnection: (connection) =>
    set((state) => ({
      connections: [...state.connections, connection],
    })),

  updateConnection: (id, updater) =>
    set((state) => ({
      connections: state.connections.map((conn) =>
        conn.id === id ? { ...conn, ...updater } : conn
      ),
    })),

  removeConnection: (id) =>
    set((state) => ({
      connections: state.connections.filter((conn) => conn.id !== id),
    })),

  // ===== SELECTION =====
  setSelection: (selection) => set({ selection }),
  clearSelection: () => set({ selection: null }),

  // ===== HOVER =====
  setHoveredElementId: (hoveredElementId) => set({ hoveredElementId }),
  setHoveredAnchor: (hoveredAnchor) => set({ hoveredAnchor }),
  setHoveredConnectionId: (hoveredConnectionId) => set({ hoveredConnectionId }),
  setHoveredLineAnchorId: (hoveredLineAnchorId) => set({ hoveredLineAnchorId }),

  // ===== MODES =====
  setMode: (mode) => set({ mode }),
  setSelectedAnchor: (selectedAnchor) => set({ selectedAnchor }),
  setMessage: (message) => set({ message }),

  startAddConnectorMode: () =>
    set({
      mode: 'add-connector',
      selectedAnchor: null,
      message: '',
    }),

  startAddAnchorMode: () =>
    set({
      mode: 'add-anchor',
      selectedAnchor: null,
      message: '',
    }),

  resetMode: () =>
    set({
      mode: null,
      selectedAnchor: null,
      hoveredAnchor: null,
      hoveredElementId: null,
      hoveredLineAnchorId: null,
      message: '',
    }),

  // ===== CONTEXT MENU =====
  openContextMenu: (position, target) =>
    set({
      contextMenu: {
        open: true,
        position,
        target,
      },
    }),

  closeContextMenu: () =>
    set({
      contextMenu: {
        open: false,
        position: null,
        target: null,
      },
    }),
}));
```

## `lib/diagram/deleteUtils.js`

```js
export function deleteConnection(connectionId, connections) {
  return connections.filter((item) => item.id !== connectionId);
}

export function deleteNodeWithLinks(nodeId, symbols, connections) {
  const nextSymbols = symbols.filter((item) => item.id !== nodeId);

  const nextConnections = connections.filter((conn) => {
    const fromSymbol = conn.from?.type === 'symbol' && conn.from.symbolId === nodeId;
    const toSymbol = conn.to?.type === 'symbol' && conn.to.symbolId === nodeId;
    return !fromSymbol && !toSymbol;
  });

  return {
    symbols: nextSymbols,
    connections: nextConnections,
  };
}
```

# 7. Центральний контролер

Оце тепер головна частина архітектури.

## `controllers/canvasController.js`

```js
import { useCanvasStore } from '@/store/useCanvasStore';
import {
  findClosestSegment,
  calculateOrthogonalPath,
  getExitDir,
  boxFromSymbol,
} from '@/lib/geometry/diagramGeometry';
import {
  deleteConnection,
  deleteNodeWithLinks,
} from '@/lib/diagram/deleteUtils';

function getAnchorPoints(element) {
  const { x, y, width, height, type } = element;

  if (type === 'rect') {
    return [
      { x: x + width / 2, y: y, direction: 'top' },
      { x: x + width / 2, y: y + height, direction: 'bottom' },
      { x, y: y + height / 2, direction: 'left' },
      { x: x + width, y: y + height / 2, direction: 'right' },
    ];
  }

  if (type === 'circle') {
    return [
      { x, y: y - height / 2, direction: 'top' },
      { x, y: y + height / 2, direction: 'bottom' },
      { x: x - width / 2, y, direction: 'left' },
      { x: x + width / 2, y, direction: 'right' },
    ];
  }

  return [];
}

function normalizeAttachment(anchor) {
  if (anchor?.kind === 'lineAnchor') {
    return {
      type: 'line',
      connectionId: anchor.parentConnectionId || anchor.connectionId,
      anchorId: anchor.id,
      segIndex: anchor.segIndex,
      t: anchor.t,
      x: anchor.x,
      y: anchor.y,
    };
  }

  return {
    type: 'symbol',
    symbolId: anchor.parentId || anchor.symbolId || anchor.id,
    direction: anchor.direction,
  };
}

function resolveAttachmentPoint(att, symbols, connections) {
  if (!att) return null;

  if (att.type === 'symbol') {
    const symbol = symbols.find((s) => s.id === att.symbolId);
    if (!symbol) return null;
    return getAnchorPoints(symbol).find((p) => p.direction === att.direction) || null;
  }

  if (att.type === 'line') {
    const parent = connections.find((c) => c.id === att.connectionId);
    if (!parent) {
      return { x: att.x, y: att.y, direction: 'mid' };
    }

    const live = (parent.additionalAnchors || []).find((a) => a.id === att.anchorId);
    if (live) {
      return { x: live.x, y: live.y, direction: 'mid' };
    }

    return { x: att.x, y: att.y, direction: 'mid' };
  }

  return null;
}

function getBoxForAttachment(att, symbols) {
  if (!att || att.type !== 'symbol') return null;
  const symbol = symbols.find((s) => s.id === att.symbolId);
  return symbol ? boxFromSymbol(symbol, 10) : null;
}

export const canvasController = {
  getAnchorPoints,

  handleStageMouseDown(e) {
    const stage = e.target.getStage();
    if (e.target === stage) {
      useCanvasStore.getState().clearSelection();
      useCanvasStore.getState().closeContextMenu();
    }
  },

  handleSymbolMouseDown(symbol) {
    useCanvasStore.getState().setSelection({
      type: 'node',
      id: symbol.id,
    });
  },

  handleSymbolDragMove(id, e) {
    const { x, y } = e.target.position();
    const store = useCanvasStore.getState();

    store.updateSymbolPosition(id, x, y);

    const symbolsNext = store.symbols.map((s) =>
      s.id === id ? { ...s, x, y } : s
    );

    const nextConnections = store.connections.map((conn) => {
      const touchesMovedSymbol =
        (conn.from?.type === 'symbol' && conn.from.symbolId === id) ||
        (conn.to?.type === 'symbol' && conn.to.symbolId === id);

      if (!touchesMovedSymbol) return conn;

      const startPoint = resolveAttachmentPoint(conn.from, symbolsNext, store.connections);
      const endPoint = resolveAttachmentPoint(conn.to, symbolsNext, store.connections);

      if (!startPoint || !endPoint) return conn;

      const dirFrom = getExitDir(conn.from, endPoint, store.connections);
      const dirTo = getExitDir(conn.to, startPoint, store.connections);

      const routing = {
        startOffset: conn.routing?.startOffset ?? 20,
        endOffset: conn.routing?.endOffset ?? 20,
      };

      const points = calculateOrthogonalPath(
        { ...startPoint, direction: dirFrom },
        { ...endPoint, direction: dirTo },
        routing,
        getBoxForAttachment(conn.from, symbolsNext),
        getBoxForAttachment(conn.to, symbolsNext),
      );

      return {
        ...conn,
        points,
        fromExitDir: dirFrom,
        toExitDir: dirTo,
      };
    });

    store.setConnections(nextConnections);
  },

  handleSymbolContextMenu(e, symbol) {
    e.evt.preventDefault();

    useCanvasStore.getState().setSelection({
      type: 'node',
      id: symbol.id,
    });

    useCanvasStore.getState().openContextMenu(
      {
        top: e.evt.clientY,
        left: e.evt.clientX,
      },
      {
        type: 'node',
        id: symbol.id,
      }
    );
  },

  handleDeleteSelected() {
    const store = useCanvasStore.getState();
    const target = store.contextMenu.target || store.selection;

    if (!target) return;

    if (target.type === 'node') {
      const result = deleteNodeWithLinks(
        target.id,
        store.symbols,
        store.connections
      );

      store.setSymbols(result.symbols);
      store.setConnections(result.connections);
      store.clearSelection();
      store.closeContextMenu();
      return;
    }

    if (target.type === 'connection') {
      const nextConnections = deleteConnection(target.id, store.connections);
      store.setConnections(nextConnections);
      store.clearSelection();
      store.closeContextMenu();
    }
  },

  handleAnchorHover(anchor) {
    useCanvasStore.getState().setHoveredAnchor(anchor);
  },

  handleElementHover(elementId) {
    useCanvasStore.getState().setHoveredElementId(elementId);
  },

  handleAnchorClick(anchor) {
    const store = useCanvasStore.getState();

    if (store.mode !== 'add-connector') return;

    if (!store.selectedAnchor) {
      store.setSelectedAnchor(anchor);
      store.setMessage("Оберіть наступну точку прив'язки");
      return;
    }

    const from = normalizeAttachment(store.selectedAnchor);
    const to = normalizeAttachment(anchor);

    const startPoint = resolveAttachmentPoint(from, store.symbols, store.connections);
    const endPoint = resolveAttachmentPoint(to, store.symbols, store.connections);

    if (!startPoint || !endPoint) {
      store.resetMode();
      store.setMessage("Не вдалося визначити точки прив'язки");
      return;
    }

    const dirFrom = getExitDir(from, endPoint, store.connections);
    const dirTo = getExitDir(to, startPoint, store.connections);

    const routing = { startOffset: 20, endOffset: 20 };

    const points = calculateOrthogonalPath(
      { ...startPoint, direction: dirFrom },
      { ...endPoint, direction: dirTo },
      routing,
      getBoxForAttachment(from, store.symbols),
      getBoxForAttachment(to, store.symbols),
    );

    const newConnection = {
      id: crypto.randomUUID(),
      from,
      to,
      points,
      routing,
      fromExitDir: dirFrom,
      toExitDir: dirTo,
      additionalAnchors: [],
    };

    store.addConnection(newConnection);
    store.resetMode();
  },

  handleConnectionClick(connectionId, event) {
    const store = useCanvasStore.getState();

    if (store.mode !== 'add-anchor') return;

    event.cancelBubble = true;

    const stage = event.target.getStage();
    const pointer = stage.getPointerPosition();
    if (!pointer) return;

    const connection = store.connections.find((c) => c.id === connectionId);
    if (!connection) return;

    const closest = findClosestSegment(connection.points, pointer.x, pointer.y);

    const newAnchor = {
      id: crypto.randomUUID(),
      x: closest.x,
      y: closest.y,
      parentConnectionId: connectionId,
      segIndex: closest.segIndex,
      t: closest.t,
      vertexIndex: null,
      kind: 'lineAnchor',
    };

    const nextConnections = store.connections.map((conn) => {
      if (conn.id !== connectionId) return conn;

      return {
        ...conn,
        additionalAnchors: [...(conn.additionalAnchors || []), newAnchor],
      };
    });

    store.setConnections(nextConnections);
    store.resetMode();
  },

  handleConnectionContextMenu(e, connection) {
    e.evt.preventDefault();

    useCanvasStore.getState().setSelection({
      type: 'connection',
      id: connection.id,
    });

    useCanvasStore.getState().openContextMenu(
      {
        top: e.evt.clientY,
        left: e.evt.clientX,
      },
      {
        type: 'connection',
        id: connection.id,
      }
    );
  },
};
```

```jsx
'use client';

import { Group, Rect, Circle } from 'react-konva';
import { useCanvasStore } from '@/store/useCanvasStore';
import { canvasController } from '@/controllers/canvasController';

function SymbolShape({ symbol }) {
  if (symbol.type === 'rect') {
    return (
      <Rect
        width={symbol.width}
        height={symbol.height}
        fill={symbol.fill}
        stroke={symbol.stroke}
        strokeWidth={symbol.strokeWidth}
      />
    );
  }

  if (symbol.type === 'circle') {
    return (
      <Circle
        x={symbol.width / 2}
        y={symbol.height / 2}
        radius={symbol.radius}
        fill={symbol.fill}
        stroke={symbol.stroke}
        strokeWidth={symbol.strokeWidth}
      />
    );
  }

  return null;
}

export default function SymbolsLayer() {
  const symbols = useCanvasStore((s) => s.symbols);
  const selection = useCanvasStore((s) => s.selection);
  const hoveredElementId = useCanvasStore((s) => s.hoveredElementId);
  const hoveredAnchor = useCanvasStore((s) => s.hoveredAnchor);
  const mode = useCanvasStore((s) => s.mode);
  const setHoveredElementId = useCanvasStore((s) => s.setHoveredElementId);
  const setHoveredAnchor = useCanvasStore((s) => s.setHoveredAnchor);

  return (
    <>
      {symbols.map((symbol) => {
        const isSelected =
          selection?.type === 'node' && selection?.id === symbol.id;

        const anchorPoints = canvasController.getAnchorPoints(symbol);

        return (
          <Group
            key={symbol.id}
            x={symbol.x}
            y={symbol.y}
            draggable
            onMouseDown={() => canvasController.handleSymbolMouseDown(symbol)}
            onDragMove={(e) => canvasController.handleSymbolDragMove(symbol.id, e)}
            onContextMenu={(e) => canvasController.handleSymbolContextMenu(e, symbol)}
            onMouseEnter={() => {
              if (mode === 'add-connector') {
                setHoveredElementId(symbol.id);
              }
            }}
            onMouseLeave={() => {
              if (mode === 'add-connector') {
                setHoveredElementId(null);
              }
            }}
          >
            <SymbolShape symbol={symbol} />

            {isSelected && (
              <Rect
                x={-6}
                y={-6}
                width={symbol.width + 12}
                height={symbol.height + 12}
                cornerRadius={12}
                stroke="#3b82f6"
                strokeWidth={1}
                dash={[6, 4]}
                listening={false}
              />
            )}

            {hoveredElementId === symbol.id &&
              anchorPoints.map((point, index) => (
                <Circle
                  key={index}
                  x={point.x - symbol.x}
                  y={point.y - symbol.y}
                  radius={6}
                  fill={
                    hoveredAnchor?.x === point.x && hoveredAnchor?.y === point.y
                      ? 'yellow'
                      : 'green'
                  }
                  stroke="black"
                  strokeWidth={1}
                  onMouseEnter={() => setHoveredAnchor(point)}
                  onMouseLeave={() => setHoveredAnchor(null)}
                  onClick={() =>
                    canvasController.handleAnchorClick({
                      ...point,
                      parentId: symbol.id,
                      kind: 'symbolAnchor',
                    })
                  }
                />
              ))}
          </Group>
        );
      })}
    </>
  );
}
```

```jsx
import { useCanvasStore } from '@/store/useCanvasStore';
import { canvasController } from '@/controllers/canvasController';

export default function ConnectionsLayer() {

  const selection = useCanvasStore((s) => s.selection);

  return (

          selection?.type === 'connection' && selection?.id === connection.id;

        return (

            <Line

              }
            />

            {(connection.additionalAnchors || []).map((anchor) => (
              <Circle
                key={anchor.id}
                x={anchor.x}
                y={anchor.y}
                radius={5}
                fill="#2563eb"
                stroke="black"
                strokeWidth={1}
                onClick={() => canvasController.handleAnchorClick(anchor)}
              />
            ))}
          </Group>
        );
      })}
    </>
  );
}
```
