# 🔥 2. ДОДАЄМО НОВУ ФУНКЦІЮ (ключова)

## `resolveDirection`

```js
export function resolveDirection(anchor, otherPoint) {
  if (anchor.direction && anchor.direction !== "auto") {
    return anchor.direction;
  }

  if (!otherPoint) return "right";

  const dx = otherPoint.x - anchor.x;
  const dy = otherPoint.y - anchor.y;

  if (Math.abs(dx) > Math.abs(dy)) {
    return dx > 0 ? "right" : "left";
  }

  return dy > 0 ? "bottom" : "top";
}
```

# 🔥 3. ДОДАЄМО RESOLVE ANCHOR

## `resolveAnchor`

```js
export function resolveAnchor(anchor, nodes, edges) {
  if (!anchor) return null;

  // NODE
  if (anchor.type === "node") {
    const node = nodes.find(n => n.id === anchor.nodeId);
    if (!node) return null;

    const { x, y } = node.position;
    const { width, height } = node.data;

    switch (anchor.direction) {
      case "top":
        return { x: x + width / 2, y, direction: "top" };
      case "bottom":
        return { x: x + width / 2, y: y + height, direction: "bottom" };
      case "left":
        return { x, y: y + height / 2, direction: "left" };
      case "right":
        return { x: x + width, y: y + height / 2, direction: "right" };
      default:
        return { x, y, direction: "right" };
    }
  }

  // EDGE
  if (anchor.type === "edge") {
    const edge = edges.find(e => e.id === anchor.edgeId);
    if (!edge) return null;

    if (anchor.segIndex != null && anchor.t != null) {
      const pts = edge.points;
      const i = anchor.segIndex;

      if (2 * i + 3 >= pts.length) return anchor;

      const x1 = pts[2 * i];
      const y1 = pts[2 * i + 1];
      const x2 = pts[2 * i + 2];
      const y2 = pts[2 * i + 3];

      return {
        x: x1 + anchor.t * (x2 - x1),
        y: y1 + anchor.t * (y2 - y1),
        direction: "auto"
      };
    }

    return {
      x: anchor.x,
      y: anchor.y,
      direction: "auto"
    };
  }

  return null;
}
```

# 🔥 4. ГОЛОВНИЙ ROUTING ENTRY POINT

👉 це заміняє ВСЕ

## `buildEdgePath`

```js
export function buildEdgePath(edge, nodes, edges) {
  const start = resolveAnchor(edge.source, nodes, edges);
  const end = resolveAnchor(edge.target, nodes, edges);

  if (!start || !end) return [];

  const dirFrom = resolveDirection(start, end);
  const dirTo = resolveDirection(end, start);

  const startBox = boxFromNode(
    nodes.find(n => n.id === edge.source.nodeId)
  );

  const endBox = boxFromNode(
    nodes.find(n => n.id === edge.target.nodeId)
  );

  let pts = calculateOrthogonalPath(
    { ...start, direction: dirFrom },
    { ...end, direction: dirTo },
    edge.data?.routing,
    startBox,
    endBox
  );

  pts = mergeCollinear(pts);
  pts = fixDiagonals(pts);

  return pts;
}
```

# 🔥 5. boxFromSymbol → boxFromNode

## ❌ було

```js
boxFromSymbol(sym)
```

## ✅ стало

```js
export function boxFromNode(node, pad = 6) {
  if (!node) return null;

  const { x, y } = node.position;
  const { width, height } = node.data;

  return {
    left: x - pad,
    right: x + width + pad,
    top: y - pad,
    bottom: y + height + pad,
  };
}
```
