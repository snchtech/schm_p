# 🔥 3. CREATE NODE FROM TOOL

## 📁 `utils/createNodeFromTool.js`

```js
import { ELEMENT_REGISTRY } from "@/components/graph/registry";

export function createNodeFromTool(type, x, y) {
  const def = ELEMENT_REGISTRY[type];
  if (!def) return null;

  return {
    id: crypto.randomUUID(),
    type,

    position: {
      x,
      y,
    },

    data: {
      width: def.preview?.w || 80,
      height: def.preview?.h || 50,
      ...def.defaults,
    },
  };
}
```

# 🔥 4. CONTROLLER — обробка кліку по Stage

## 📁 `canvasController.js`

```js
import { useCanvasStore } from "@/store/useCanvasStore";
import { createNodeFromTool } from "@/utils/createNodeFromTool";

handleStageMouseDown(e) {
  const store = useCanvasStore.getState();

  const activeTool = store.activeTool;
  if (!activeTool) return;

  const stage = e.target.getStage();

  // 👉 тільки якщо клік по пустому canvas
  if (e.target !== stage) return;

  const pointer = stage.getPointerPosition();
  if (!pointer) return;

  const node = createNodeFromTool(
    activeTool,
    pointer.x,
    pointer.y
  );

  if (!node) return;

  store.addNode(node);
  store.resetActiveTool(); // 👈 після вставки
}
```

# 🔥 6. ВАЖЛИВО: КООРДИНАТИ ПРИ ZOOM

👉 у тебе вже є zoom/pan → тому pointer треба нормалізувати

---

## 🔥 helper

```js
export function getRelativePointer(stage, layer) {
  const pointer = stage.getPointerPosition();
  if (!pointer) return null;

  const scale = layer.scaleX();
  const pos = layer.position();

  return {
    x: (pointer.x - pos.x) / scale,
    y: (pointer.y - pos.y) / scale,
  };
}
```

---

## 🔥 використовуємо в controller

```js
import { getRelativePointer } from "@/utils/coords";

const pointer = getRelativePointer(stage, store.layerRef);
```
