## 2. 🧠 Zustand store (це ключове)

Ти абсолютно правильно подумав — **Zustand тут ідеально підходить**

### Мінімальний store:

```js
import { create } from 'zustand'

export const useCanvasStore = create((set, get) => ({
  // ===== STATE =====
  symbols: [],
  connections: [],
  selection: null,

  mode: null, // 'add-connector' | 'add-anchor'

  hovered: {
    elementId: null,
    anchor: null,
  },

  // ===== ACTIONS =====
  setSymbols: (symbols) => set({ symbols }),

  addSymbol: (symbol) =>
    set((state) => ({ symbols: [...state.symbols, symbol] })),

  updateSymbolPosition: (id, x, y) =>
    set((state) => ({
      symbols: state.symbols.map((s) =>
        s.id === id ? { ...s, x, y } : s
      ),
    })),

  setSelection: (selection) => set({ selection }),

  clearSelection: () => set({ selection: null }),

  setMode: (mode) => set({ mode }),

}))
```

## 3. 🎯 Central Controller (НАЙВАЖЛИВІШЕ)

👉 Це те, чого у тебе зараз немає

---

### 📁 `/controllers/canvasController.js`

```js
import { useCanvasStore } from "../store/useCanvasStore";

export const canvasController = {
  handleStageMouseDown: (e) => {
    const { mode } = useCanvasStore.getState();

    const stage = e.target.getStage();

    // клік по пустому місцю
    if (e.target === stage) {
      useCanvasStore.getState().clearSelection();
    }
  },

  handleElementClick: (element) => {
    useCanvasStore.getState().setSelection({
      type: "node",
      id: element.id,
    });
  },

  handleDragMove: (id, e) => {
    const { x, y } = e.target.position();

    useCanvasStore
      .getState()
      .updateSymbolPosition(id, x, y);
  },

  handleContextMenu: (e, element) => {
    e.evt.preventDefault();

    useCanvasStore.getState().setContextMenu({
      open: true,
      position: {
        x: e.evt.clientX,
        y: e.evt.clientY,
      },
      target: element,
    });
  },
};
```
