# 🔥 1. Store — nodeRefs

## `useCanvasStore.js`

```js
nodeRefs: {},

registerNodeRef: (id, ref) =>
  set((state) => ({
    nodeRefs: {
      ...state.nodeRefs,
      [id]: ref,
    },
  })),

unregisterNodeRef: (id) =>
  set((state) => {
    const next = { ...state.nodeRefs };
    delete next[id];
    return { nodeRefs: next };
  }),
```

# 🔥 3. Новий компонент: `SelectionTransformer.jsx`

## 📁 `components/SchemeEditor/SelectionTransformer.jsx`

```jsx
"use client";

import { useEffect, useRef } from "react";
import { Transformer } from "react-konva";
import { useCanvasStore } from "@/store/useCanvasStore";

export default function SelectionTransformer() {
  const transformerRef = useRef(null);

  const selection = useCanvasStore((s) => s.selection);
  const nodeRefs = useCanvasStore((s) => s.nodeRefs);

  useEffect(() => {
    const transformer = transformerRef.current;
    if (!transformer) return;

    // 🔥 MULTI SELECT support
    const selectedIds =
      selection?.type === "nodes"
        ? selection.ids
        : selection?.type === "node"
        ? [selection.id]
        : [];

    const nodes = selectedIds
      .map((id) => nodeRefs[id])
      .filter(Boolean);

    transformer.nodes(nodes);
    transformer.getLayer()?.batchDraw();
  }, [selection, nodeRefs]);

  return (
    <Transformer
      ref={transformerRef}
      rotateEnabled={false}
      anchorSize={8}
      borderStroke="#3b82f6"
      anchorFill="#3b82f6"
      anchorStroke="white"
      anchorStrokeWidth={1}
    />
  );
}
```

# 🔥 5. Resize логіка (CRITICAL)

Transformer сам міняє scale, але нам треба:

👉 записати нові width/height в store

---

## додаємо в `SelectionTransformer`

```jsx
onTransformEnd={() => {
  const store = useCanvasStore.getState();
  const transformer = transformerRef.current;
  const nodes = transformer.nodes();

  nodes.forEach((nodeRef) => {
    const id = Object.keys(store.nodeRefs).find(
      (key) => store.nodeRefs[key] === nodeRef
    );

    if (!id) return;

    const scaleX = nodeRef.scaleX();
    const scaleY = nodeRef.scaleY();

    const node = store.nodes.find((n) => n.id === id);
    if (!node) return;

    const newWidth = Math.max(20, node.data.width * scaleX);
    const newHeight = Math.max(20, node.data.height * scaleY);

    // reset scale
    nodeRef.scaleX(1);
    nodeRef.scaleY(1);

    store.updateNodeSize(id, newWidth, newHeight);
  });
}}
```

# 🔥 7. Оновлення edges при resize

👉 ДУЖЕ ВАЖЛИВО

після resize — edges треба перерахувати

---

## додаємо після updateNodeSize:

```js
const nextNodes = store.nodes.map((n) =>
  n.id === id
    ? {
        ...n,
        data: { ...n.data, width, height },
      }
    : n
);

const nextEdges = store.edges.map((edge) => {
  if (
    edge.source.nodeId !== id &&
    edge.target.nodeId !== id
  ) return edge;

  return {
    ...edge,
    points: buildEdgePath(edge, nextNodes, store.edges),
  };
});

store.setEdges(nextEdges);
```
