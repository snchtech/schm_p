---

# controllers/canvasController.js

---

## ДОДАЙ

```js id="8mptkj"
handleConnectorContextMenu(e, connector) {

   e.evt.preventDefault()

   const store = useCanvasStore.getState()

   const hovered =
      store.hoveredConnectorSegment

   store.setContextMenu({

      mouseX: e.evt.clientX,
      mouseY: e.evt.clientY,

      connectorId: connector.id,

      segmentId:
         hovered?.connectorId === connector.id
            ? hovered.segmentId
            : null,

      type: "connector",
   })
}
```

# 4. ДОДАЙ CREATE PATH HANDLE ACTION

---

## ДОДАЙ

```js id="9bjlwm"
addPathHandleToConnector(
   connectorId,
   segmentId,
) {

   const store = useCanvasStore.getState()

   const connector =
      store.edges.find(
         e => e.id === connectorId
      )

   if (!connector) return

   if (!segmentId) return

   const handle =
      createPathHandleAtSegment(
         connector,
         segmentId,
      )

   if (!handle) return

   const next =
      addConnectorPathHandle(
         connector,
         handle,
      )

   store.replaceEdge(
      connector.id,
      rebuildConnectorGeometry(
         next,
         store.nodes,
      )
   )

   store.clearContextMenu()
}
```

# 6. ТЕПЕР MUI MENU

---

# components/ConnectorContextMenu.jsx

---

## СТВОРИ НОВИЙ COMPONENT

```jsx id="hfef3f"
"use client"

import React from "react"

import {
   Menu,
   MenuItem,
   Divider,
} from "@mui/material"

import { useCanvasStore }
from "@/store/useCanvasStore"

import { canvasController }
from "@/controllers/canvasController"

export default function ConnectorContextMenu() {

   const contextMenu =
      useCanvasStore(
         s => s.contextMenu
      )

   const clearContextMenu =
      useCanvasStore(
         s => s.clearContextMenu
      )

   const open =
      contextMenu?.type === "connector"

   return (
      <Menu
         open={open}

         onClose={() => {
            clearContextMenu()
         }}

         anchorReference="anchorPosition"

         anchorPosition={
            open
               ? {
                    top: contextMenu.mouseY,
                    left: contextMenu.mouseX,
                 }
               : undefined
         }
      >

         <MenuItem
            onClick={() => {

               canvasController
                  .addPathHandleToConnector(

                     contextMenu.connectorId,
                     contextMenu.segmentId,
                  )
            }}
         >
            Додати точку шляху
         </MenuItem>

         <Divider />

         <MenuItem
            sx={{ color: "error.main" }}

            onClick={() => {

               canvasController
                  .deleteConnector(
                     contextMenu.connectorId
                  )
            }}
         >
            Видалити конектор
         </MenuItem>

      </Menu>
   )
}
```
