# keyboardManager.js

```js
import { useCanvasStore } from './useCanvasStore'

export function initKeyboardManager() {
   const down = (e) => {
      const store = useCanvasStore.getState()

      if (e.key === 'Shift') {
         store.setKeyboardState({ shift: true })
      }

      if (e.key === 'Control') {
         store.setKeyboardState({ ctrl: true })
      }

      if (e.key === 'Alt') {
         store.setKeyboardState({ alt: true })
      }
   }

   const up = (e) => {
      const store = useCanvasStore.getState()

      if (e.key === 'Shift') {
         store.setKeyboardState({ shift: false })
      }

      if (e.key === 'Control') {
         store.setKeyboardState({ ctrl: false })
      }

      if (e.key === 'Alt') {
         store.setKeyboardState({ alt: false })
      }
   }

   window.addEventListener('keydown', down)
   window.addEventListener('keyup', up)

   return () => {
      window.removeEventListener('keydown', down)
      window.removeEventListener('keyup', up)
   }
}
```
