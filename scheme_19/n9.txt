```js id="jlwm99"
export function serializeCanvas({
   nodes,
   edges,
}) {

   return JSON.stringify({

      schemaVersion:
         CURRENT_SCHEMA_VERSION,

      meta: {

         savedAt:
            new Date().toISOString(),

         app: "etls-scheme-editor",

         routerVersion:
            CURRENT_ROUTER_VERSION,
      },

      nodes: nodes.map(serializeNode),

      edges: edges.map(serializeEdge),
   })
}
```

# 4. SERIALIZE EDGE

---

## ДОДАЙ

```js id="jlwm92"
function serializeEdge(edge) {

   return {

      id: edge.id,

      kind: edge.kind,

      connectorType:
         edge.connectorType,

      pathType:
         edge.pathType,

      start: edge.start,

      end: edge.end,

      route: {

         vertices:
            edge.route?.vertices || [],

         pathHandles:
            edge.route?.pathHandles || [],

         terminalDirections:
            edge.route
               ?.terminalDirections || {

                  start: "horizontal",
                  end: "horizontal",
               },

         routing: {

            routerVersion:
               CURRENT_ROUTER_VERSION,

            preserveTopology: true,

            preserveElbows: true,
         },
      },

      style: edge.style || {},

      endcaps:
         edge.endcaps || {},
   }
}
```

# 8. DESERIALIZE

---

## ДОДАЙ

```js id="jlwm96"
from "@/lib/connectors/connectorGeometry"
```

---

## І

```js id="jlwm97"
export function deserializeCanvas(
   json,
) {

   let data =
      JSON.parse(json)

   const version =
      data.schemaVersion || 1

   const migration =
      migrations[version]

   if (migration) {
      data = migration(data)
   }

   const rebuiltEdges =
      data.edges.map(edge => {

         if (edge.kind !== "connector") {
            return edge
         }

         return rebuildConnectorGeometry(
            edge,
            data.nodes,
         )
      })

   return {

      nodes: data.nodes,

      edges: rebuiltEdges,
   }
}
```

# 10. SAVE BUTTON

---

# components/SaveCanvasButton.jsx

---

## СТВОРИ

```jsx id="jlwm101"
"use client"

import Button
from "@mui/material/Button"

import { useCanvasStore }
from "@/store/useCanvasStore"

import {
   serializeCanvas,
}
from "@/lib/schemePersistence"

export default function SaveCanvasButton() {

   const nodes =
      useCanvasStore(s => s.nodes)

   const edges =
      useCanvasStore(s => s.edges)

   function handleSave() {

      const json =
         serializeCanvas({
            nodes,
            edges,
         })

      localStorage.setItem(
         "scheme_v1",
         json,
      )

      console.log(
         "SCHEME SAVED"
      )
   }

   return (

      <Button
         variant="contained"
         onClick={handleSave}
      >
         Save Scheme
      </Button>
   )
}
```

# 11. LOAD BUTTON

---

# components/LoadCanvasButton.jsx

---

## СТВОРИ

```jsx id="jlwm102"
"use client"

import Button
from "@mui/material/Button"

import { useCanvasStore }
from "@/store/useCanvasStore"

import {
   deserializeCanvas,
}
from "@/lib/schemePersistence"

export default function LoadCanvasButton() {

   const setCanvasData =
      useCanvasStore(
         s => s.setCanvasData
      )

   function handleLoad() {

      const raw =
         localStorage.getItem(
            "scheme_v1"
         )

      if (!raw) {
         return
      }

      const data =
         deserializeCanvas(raw)

      setCanvasData(data)

      console.log(
         "SCHEME LOADED"
      )
   }

   return (

      <Button
         variant="outlined"
         onClick={handleLoad}
      >
         Load Scheme
      </Button>
   )
}
```
