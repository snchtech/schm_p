Добре. Тоді дам тобі повний refactor текстом.
Це буде вже правильна topology-based модель для `pathHandles`, яка:

* НЕ буде ламатись після normalize
* НЕ буде залежати від runtime segment ids
* буде стабільно працювати з drag/rebuild
* сумісна з твоєю новою архітектурою internal vertices.

---

# 1. ГОЛОВНА ПРОБЛЕМА ЗАРАЗ

Зараз у тебе:

```js
pathHandle.parentSegmentId
```

але `segmentId`:

* runtime
* rebuild-иться
* normalize-иться
* змінюється після drag.

Тому:

* handles "відвалюються"
* segment match ламається
* topology стає нестабільною.

# НОВА ВЕРСІЯ

```js
export function createPathHandleAtSegment(
  connector,
  segmentId,
) {
  const segments = connector.geometry?.segmentsMeta || []

  const seg = segments.find(
    (s) => s.id === segmentId
  )

  if (!seg) return null

  return {
    id: crypto.randomUUID(),

    beforeVertexId: seg.sourceFrom,
    afterVertexId: seg.sourceTo,

    axis:
      seg.orientation === "horizontal"
        ? "y"
        : "x",

    offset: 0,
    depth: 20,

    x: seg.cx,
    y: seg.cy,
  }
}
```

# 8. НОВИЙ HELPER

---

## ДОДАЙ

```js
function findSegmentIndexByVertexPair(
  fullRoute,
  beforeId,
  afterId,
) {
  for (let i = 0; i < fullRoute.length - 1; i++) {

    const a = fullRoute[i]
    const b = fullRoute[i + 1]

    if (
      a.id === beforeId &&
      b.id === afterId
    ) {
      return i
    }
  }

  return -1
}
```

# 9. ПЕРЕПИСУЄМО insertPathHandleIntoSegmentById

---

# ЗАМІНИ НА:

```js
export function insertPathHandleIntoRoute(
  routeBuild,
  pathHandle,
) {
  if (!routeBuild || !pathHandle) {
    return routeBuild
  }

  const full = []

  for (let i = 0; i < routeBuild.points.length; i += 2) {
    full.push({
      x: routeBuild.points[i],
      y: routeBuild.points[i + 1],
    })
  }

  const topology =
    routeBuild.topologyPoints || []

  const segmentIndex =
    findSegmentIndexByVertexPair(
      topology,
      pathHandle.beforeVertexId,
      pathHandle.afterVertexId,
    )

  if (segmentIndex < 0) {
    return routeBuild
  }

  const flatIndex = segmentIndex * 2

  const x1 = full[segmentIndex].x
  const y1 = full[segmentIndex].y

  const x2 = full[segmentIndex + 1].x
  const y2 = full[segmentIndex + 1].y

  const before =
    routeBuild.points.slice(
      0,
      flatIndex + 2,
    )

  const after =
    routeBuild.points.slice(
      flatIndex + 2,
    )

  let inserted = []

  const half = pathHandle.depth || 20

  if (pathHandle.axis === "y") {

    const leftX =
      pathHandle.x - half

    const rightX =
      pathHandle.x + half

    inserted = [
      leftX, y1,
      leftX, pathHandle.y,

      rightX, pathHandle.y,
      rightX, y1,
    ]
  }

  if (pathHandle.axis === "x") {

    const topY =
      pathHandle.y - half

    const bottomY =
      pathHandle.y + half

    inserted = [
      x1, topY,
      pathHandle.x, topY,

      pathHandle.x, bottomY,
      x1, bottomY,
    ]
  }

  const nextPoints =
    normalizePoints([
      ...before,
      ...inserted,
      ...after,
    ])

  return {
    ...routeBuild,

    points: nextPoints,
  }
}
```
