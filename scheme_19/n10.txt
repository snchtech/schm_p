# ПРИКЛАД

```js id="jlwm220"
export function rebuildConnectorGeometryByVersion(
   edge,
   nodes,
) {

   const version =
      edge.route?.routing?.routerVersion || 1

   switch(version) {

      case 1:
         return rebuildConnectorGeometryV1(
            edge,
            nodes,
         )

      case 2:
         return rebuildConnectorGeometryV2(
            edge,
            nodes,
         )

      default:
         return rebuildConnectorGeometryV1(
            edge,
            nodes,
         )
   }
}
```
