# 3. Додаємо helper-и в `connectorEditing.js`

Потрібно:

* створити waypoint у середині цього сегмента

```js id="cbs8qv"
function same(a, b, eps = 0.001) {
  return Math.abs(a - b) < eps;
}

export function getConnectorSegments(points = []) {
  const segments = [];

  for (let i = 0; i < points.length - 2; i += 2) {
    const x1 = points[i];
    const y1 = points[i + 1];
    const x2 = points[i + 2];
    const y2 = points[i + 3];

    let orientation = "diagonal";
    if (same(y1, y2)) orientation = "horizontal";
    if (same(x1, x2)) orientation = "vertical";

    segments.push({
      index: i / 2,
      x1,
      y1,
      x2,
      y2,
      orientation,
      cx: (x1 + x2) / 2,
      cy: (y1 + y2) / 2,
    });
  }

  return segments;
}

export function getConnectorSegmentHandles(connector) {
  if (connector.connectorType !== "orthogonal-elbow") {
    return [];
  }

  const points = connector.geometry?.points || [];
  const segments = getConnectorSegments(points);

  return segments
    .filter(
      (seg) => seg.orientation === "horizontal" || seg.orientation === "vertical"
    )
    .map((seg) => ({
      id: `seg-handle-${connector.id}-${seg.index}`,
      connectorId: connector.id,
      segmentIndex: seg.index,
      orientation: seg.orientation,
      x: seg.cx,
      y: seg.cy,
    }));
}

function projectPointOnSegment(px, py, seg) {
  const { x1, y1, x2, y2, orientation } = seg;

  if (orientation === "horizontal") {
    const minX = Math.min(x1, x2);
    const maxX = Math.max(x1, x2);
    const x = Math.max(minX, Math.min(maxX, px));
    const y = y1;
    const dx = px - x;
    const dy = py - y;
    return {
      x,
      y,
      distance: Math.sqrt(dx * dx + dy * dy),
    };
  }

  if (orientation === "vertical") {
    const minY = Math.min(y1, y2);
    const maxY = Math.max(y1, y2);
    const x = x1;
    const y = Math.max(minY, Math.min(maxY, py));
    const dx = px - x;
    const dy = py - y;
    return {
      x,
      y,
      distance: Math.sqrt(dx * dx + dy * dy),
    };
  }

  // fallback для safety
  const mx = (x1 + x2) / 2;
  const my = (y1 + y2) / 2;
  const dx = px - mx;
  const dy = py - my;

  return {
    x: mx,
    y: my,
    distance: Math.sqrt(dx * dx + dy * dy),
  };
}

export function findClosestConnectorSegment(connector, point) {
  const points = connector.geometry?.points || [];
  const segments = getConnectorSegments(points);

  let best = null;
  let bestDistance = Infinity;

  for (const seg of segments) {
    if (seg.orientation !== "horizontal" && seg.orientation !== "vertical") {
      continue;
    }

    const projected = projectPointOnSegment(point.x, point.y, seg);

    if (projected.distance < bestDistance) {
      bestDistance = projected.distance;
      best = {
        segmentIndex: seg.index,
        orientation: seg.orientation,
        x: projected.x,
        y: projected.y,
      };
    }
  }

  return best;
}

export function createWaypointAtSegment(connector, segmentIndex) {
  const points = connector.geometry?.points || [];
  const i = segmentIndex * 2;

  if (i + 3 >= points.length) return null;

  const x1 = points[i];
  const y1 = points[i + 1];
  const x2 = points[i + 2];
  const y2 = points[i + 3];

  return {
    id: crypto.randomUUID(),
    x: (x1 + x2) / 2,
    y: (y1 + y2) / 2,
  };
}

/**
 * MVP:
 * - якщо waypoint ще нема -> створюємо 1 waypoint
 * - якщо waypoint є -> рухаємо перший waypoint
 */
export function moveElbowSegmentHandle(connector, handle, pos) {
  const waypoints = [...(connector.route?.waypoints || [])];

  if (!waypoints.length) {
    waypoints.push({
      id: crypto.randomUUID(),
      x: handle.x,
      y: handle.y,
    });
  }

  const wp = waypoints[0];

  if (handle.orientation === "horizontal") {
    wp.y = pos.y;
  } else if (handle.orientation === "vertical") {
    wp.x = pos.x;
  }

  return {
    ...connector,
    route: {
      ...connector.route,
      waypoints,
    },
  };
}
```

```js id="qvu8w9"
      ...connector.route,
      waypoints: [...(connector.route?.waypoints || []), waypoint],
    },

  return {
    ...connector,
```

## Методи для context menu

```js id="3jv4j6"
handleConnectorContextMenu(e, connector) {
  e.evt.preventDefault();
  e.cancelBubble = true;

  const store = useCanvasStore.getState();
  const stage = e.target.getStage();
  const pointer = stage?.getPointerPosition();

  if (!pointer) return;

  const closest = findClosestConnectorSegment(connector, {
    x: pointer.x,
    y: pointer.y,
  });

  store.setSelection?.({ type: "edge", id: connector.id });

  store.setContextMenu?.({
    type: "connector",
    edgeId: connector.id,
    segmentIndex: closest?.segmentIndex ?? null,
    x: pointer.x,
    y: pointer.y,
  });
},

handleConnectorWaypointContextMenu(e, connector, waypoint) {
  e.evt.preventDefault();
  e.cancelBubble = true;

  const store = useCanvasStore.getState();
  const stage = e.target.getStage();
  const pointer = stage?.getPointerPosition();

  if (!pointer) return;

  store.setSelection?.({ type: "edge", id: connector.id });

  store.setContextMenu?.({
    type: "connector-waypoint",
    edgeId: connector.id,
    waypointId: waypoint.id,
    x: pointer.x,
    y: pointer.y,

},
```

## Додати waypoint

```js id="t8p0sx"
addConnectorWaypointFromContextMenu() {
  const store = useCanvasStore.getState();
  const menu = store.contextMenu;

  if (!menu || menu.type !== "connector") return;

  const connector = store.edges.find((e) => e.id === menu.edgeId);
  if (!connector || connector.kind !== "connector") return;
  if (connector.connectorType !== "orthogonal-elbow") return;
  if (menu.segmentIndex == null) return;

  const waypoint = createWaypointAtSegment(connector, menu.segmentIndex);
  if (!waypoint) {
    store.clearContextMenu?.();
    return;
  }

  let next = addConnectorWaypoint(connector, waypoint);
  next = rebuildConnectorGeometry(next, store.nodes);

  store.replaceEdge?.(connector.id, next);
  store.clearContextMenu?.();
},
```

# 6. Додаємо waypoint render у `ConnectorsLayer.jsx`

Тепер, коли connector selected, показуємо ще й самі waypoint-и.

Вони:

* показуються тільки у вибраного connector-а
* правий клік по waypoint -> “Прибрати точку шляху”

Онови `ConnectorsLayer.jsx` так.

```jsx id="o3fe98"
"use client";

import React from "react";
import { Group, Line, Circle, Rect } from "react-konva";
import { useCanvasStore } from "@/store/useCanvasStore";
import { canvasController } from "@/controllers/canvasController";
import { getConnectorSegmentHandles } from "@/lib/connectors/connectorEditing";

export default function ConnectorsLayer() {
  const edges = useCanvasStore((s) => s.edges);
  const selection = useCanvasStore((s) => s.selection);
  const setSelection = useCanvasStore((s) => s.setSelection);

  const connectors = edges.filter((edge) => edge.kind === "connector");

  return (
    <>
      {connectors.map((connector) => {
        const pts = connector.geometry?.points || [];
        if (!pts || pts.length < 4) return null;

        const isSelected =
          selection?.type === "edge" && selection?.id === connector.id;

        const startPoint = {
          x: pts[0],
          y: pts[1],
        };

        const endPoint = {
          x: pts[pts.length - 2],
          y: pts[pts.length - 1],
        };

        const segmentHandles = isSelected
          ? getConnectorSegmentHandles(connector)
          : [];

        const waypoints = isSelected
          ? connector.route?.waypoints || []
          : [];

        return (
          <Group key={connector.id}>
            <Line
              points={pts}
              stroke={isSelected ? "#2F80ED" : connector.style?.stroke || "black"}
              strokeWidth={
                isSelected
                  ? Math.max((connector.style?.strokeWidth || 2) + 1, 3)
                  : connector.style?.strokeWidth || 2
              }
              lineJoin="round"
              lineCap="round"
              onMouseDown={(e) => {
                e.cancelBubble = true;
                setSelection({ type: "edge", id: connector.id });
              }}
              onClick={(e) => {
                e.cancelBubble = true;
                setSelection({ type: "edge", id: connector.id });
              }}
              onContextMenu={(e) => {
                canvasController.handleConnectorContextMenu?.(e, connector);
              }}
            />

            {isSelected && (
              <>
                {/* start */}
                <Circle
                  x={startPoint.x}
                  y={startPoint.y}
                  radius={6}
                  fill="white"
                  stroke="#2F80ED"
                  strokeWidth={2}
                  draggable
                  onDragMove={(e) => {
                    canvasController.handleConnectorStartDragMove?.(
                      connector.id,
                      e
                    );
                  }}
                  onDragEnd={(e) => {
                    canvasController.handleConnectorStartDragEnd?.(
                      connector.id,
                      e
                    );
                  }}
                />

                {/* end */}
                <Circle
                  x={endPoint.x}
                  y={endPoint.y}
                  radius={6}
                  fill="white"
                  stroke="#2F80ED"
                  strokeWidth={2}
                  draggable
                  onDragMove={(e) => {
                    canvasController.handleConnectorEndDragMove?.(
                      connector.id,
                      e
                    );
                  }}
                  onDragEnd={(e) => {
                    canvasController.handleConnectorEndDragEnd?.(
                      connector.id,
                      e
                    );
                  }}
                />

                {/* segment handles */}
                {segmentHandles.map((handle) => (
                  <Rect
                    key={handle.id}
                    x={handle.x - 5}
                    y={handle.y - 5}
                    width={10}
                    height={10}
                    fill="#2D9CDB"
                    stroke="black"
                    strokeWidth={1}
                    draggable
                    dragBoundFunc={(pos) =>
                      handle.orientation === "horizontal"
                        ? { x: handle.x - 5, y: pos.y }
                        : { x: pos.x, y: handle.y - 5 }
                    }
                    onDragMove={(e) => {
                      canvasController.handleConnectorSegmentDragMove?.(
                        connector.id,
                        handle,
                        e
                      );
                    }}
                    onDragEnd={(e) => {
                      canvasController.handleConnectorSegmentDragEnd?.(
                        connector.id,
                        handle,
                        e
                      );
                    }}
                  />
                ))}

                {/* waypoints */}
                {waypoints.map((wp) => (
                  <Circle
                    key={wp.id}
                    x={wp.x}
                    y={wp.y}
                    radius={5}
                    fill="#F2C94C"
                    stroke="black"
                    strokeWidth={1}
                    onContextMenu={(e) => {
                      canvasController.handleConnectorWaypointContextMenu?.(
                        e,
                        connector,
                        wp
                      );
                    }}
                  />
                ))}
              </>
            )}
          </Group>
        );
      })}
    </>
  );
}
```

# 7. Компонент меню `CanvasContextMenu.jsx`

Якщо в тебе вже є загальний menu-компонент — просто додай 2 пункти.
Якщо ні, ось готовий мінімальний варіант.

```jsx id="kjwmln"
"use client";

import React from "react";
import { Menu, MenuItem } from "@mui/material";
import { useCanvasStore } from "@/store/useCanvasStore";
import { canvasController } from "@/controllers/canvasController";

export default function CanvasContextMenu() {
  const contextMenu = useCanvasStore((s) => s.contextMenu);
  const clearContextMenu = useCanvasStore((s) => s.clearContextMenu);

  return (
    <Menu
      open={!!contextMenu}
      onClose={clearContextMenu}
      anchorReference="anchorPosition"
      anchorPosition={
        contextMenu
          ? {
              top: contextMenu.y,
              left: contextMenu.x,
            }
          : undefined
      }
    >
      {contextMenu?.type === "connector" && (
        <MenuItem
          onClick={() => {
            canvasController.addConnectorWaypointFromContextMenu?.();
          }}
        >
          Додати точку шляху
        </MenuItem>
      )}

      {contextMenu?.type === "connector-waypoint" && (
        <MenuItem
          onClick={() => {
            canvasController.removeConnectorWaypointFromContextMenu?.();
          }}
        >
          Прибрати точку шляху
        </MenuItem>
      )}
    </Menu>
  );
}
```

# 10. Важливе чесне уточнення

У поточному MVP:

* `moveElbowSegmentHandle()` поки працює через перший waypoint
* тобто при багатьох waypoint логіка ще не ідеальна
* але **додавання/видалення waypoint** уже буде працювати як корисний перший крок

Наступним етапом треба буде зробити:

* коректне зіставлення конкретного segment handle до конкретного waypoint
* drag самих waypoint-ів
* rebuild із кількома точками шляху вже без спрощень
