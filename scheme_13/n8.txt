Добре. Тоді наступний крок — **segment handles** для `orthogonal-elbow`.

Це саме ті сині квадратні точки по центру сегментів, які:

* показуються тільки коли connector вибраний
* для горизонтального сегмента рухаються тільки по `y`
* для вертикального — тільки по `x`
* змінюють форму лінії
* не чіпають поки старий `EdgesLayer`

Нижче даю тобі **MVP реалізацію**, яка лягає на той каркас, що ми вже зробили.

# 1. Важлива архітектурна думка

Для першого робочого варіанту я раджу таку модель:

* `geometry.points` — це фінальна polyline
* `route.waypoints` — поки що керуючі точки
* **segment handle drag** буде змінювати nearest/internal waypoint
* якщо waypoint ще нема — будемо створювати базовий waypoint

Але щоб не ускладнювати зараз модель занадто сильно, для MVP можна зробити простіше:

## Для `orthogonal-elbow`

вважати, що:

* якщо `waypoints.length === 0`, то лінія має один базовий “злам”
* segment handle drag створює/оновлює waypoint
* далі geometry перебудовується

Тобто ми не редагуємо напряму `geometry.points`, а редагуємо `route.waypoints`.

Це правильний напрям.

function same(a, b, eps = 0.001) {
  return Math.abs(a - b) < eps;
}

export function getNodeAnchorPosition(node, anchor = "right") {
  if (!node) return null;

  const { x, y } = node.position;
  const width = node.data?.width || 0;
  const height = node.data?.height || 0;

  switch (anchor) {
    case "top":
      return { x: x + width / 2, y };
    case "bottom":
      return { x: x + width / 2, y: y + height };
    case "left":
      return { x, y: y + height / 2 };
    case "right":
    default:
      return { x: x + width, y: y + height / 2 };
  }
}

export function resolveConnectorEndpoint(endpoint, nodes) {
  if (!endpoint) return { x: 0, y: 0 };

  if (endpoint.attached?.type === "node") {
    const node = nodes.find((n) => n.id === endpoint.attached.nodeId);
    const pos = getNodeAnchorPosition(node, endpoint.attached.anchor);
    if (pos) return pos;
  }

  return {
    x: endpoint.x,
    y: endpoint.y,
  };
}

export function normalizePoints(points = []) {
  if (points.length <= 4) return points;

  const raw = [];
  for (let i = 0; i < points.length; i += 2) {
    raw.push({ x: points[i], y: points[i + 1] });
  }

  const dedup = [];
  for (const p of raw) {
    const last = dedup[dedup.length - 1];
    if (!last || !same(last.x, p.x) || !same(last.y, p.y)) {
      dedup.push(p);
    }
  }

  if (dedup.length <= 2) {
    return dedup.flatMap((p) => [p.x, p.y]);
  }

  const out = [dedup[0]];

  for (let i = 1; i < dedup.length - 1; i++) {
    const a = out[out.length - 1];
    const b = dedup[i];
    const c = dedup[i + 1];

    const horizontal = same(a.y, b.y) && same(b.y, c.y);
    const vertical = same(a.x, b.x) && same(b.x, c.x);

    if (horizontal || vertical) continue;

    out.push(b);
  }

  out.push(dedup[dedup.length - 1]);

  return out.flatMap((p) => [p.x, p.y]);
}

export function buildOrthogonalStraightPoints(start, end) {
  if (same(start.x, end.x) || same(start.y, end.y)) {
    return [start.x, start.y, end.x, end.y];
  }

  return [start.x, start.y, end.x, start.y, end.x, end.y];
}

/**
 * orthogonal-elbow:
 * - 0 waypoint -> одна L-форма
 * - 1+ waypoint -> йдемо сходинками через них
 */
export function buildOrthogonalElbowPoints(start, end, waypoints = []) {
  const points = [start.x, start.y];

  if (!waypoints.length) {
    points.push(end.x, start.y, end.x, end.y);
    return normalizePoints(points);
  }

  let current = { ...start };

  for (const wp of waypoints) {
    if (!same(current.x, wp.x)) {
      points.push(wp.x, current.y);
      current = { x: wp.x, y: current.y };
    }

    if (!same(current.y, wp.y)) {
      points.push(current.x, wp.y);
      current = { x: current.x, y: wp.y };
    }
  }

  if (!same(current.x, end.x)) {
    points.push(end.x, current.y);
    current = { x: end.x, y: current.y };
  }

  if (!same(current.y, end.y)) {
    points.push(current.x, end.y);
  }

  return normalizePoints(points);
}

export function buildConnectorPoints(connector, nodes) {
  const start = resolveConnectorEndpoint(connector.start, nodes);
  const end = resolveConnectorEndpoint(connector.end, nodes);
  const waypoints = connector.route?.waypoints || [];

  switch (connector.connectorType) {
    case "orthogonal-elbow":
      return buildOrthogonalElbowPoints(start, end, waypoints);

    case "orthogonal-straight":
    default:
      return buildOrthogonalStraightPoints(start, end);
  }
}

export function rebuildConnectorGeometry(connector, nodes) {
  return {
    ...connector,
    geometry: {
      ...connector.geometry,
      points: buildConnectorPoints(connector, nodes),
    },
  };
}

function same(a, b, eps = 0.001) {
  return Math.abs(a - b) < eps;
}

export function getConnectorSegments(points = []) {
  const segments = [];

  for (let i = 0; i < points.length - 2; i += 2) {
    const x1 = points[i];
    const y1 = points[i + 1];
    const x2 = points[i + 2];
    const y2 = points[i + 3];

    let orientation = "diagonal";
    if (same(y1, y2)) orientation = "horizontal";
    if (same(x1, x2)) orientation = "vertical";

    segments.push({
      index: i / 2,
      x1,
      y1,
      x2,
      y2,
      orientation,
      cx: (x1 + x2) / 2,
      cy: (y1 + y2) / 2,
    });
  }

  return segments;
}

export function getConnectorSegmentHandles(connector) {
  if (connector.connectorType !== "orthogonal-elbow") {
    return [];
  }

  const points = connector.geometry?.points || [];
  const segments = getConnectorSegments(points);

  return segments
    .filter(
      (seg) => seg.orientation === "horizontal" || seg.orientation === "vertical"
    )
    .map((seg) => ({
      id: `seg-handle-${connector.id}-${seg.index}`,
      connectorId: connector.id,
      segmentIndex: seg.index,
      orientation: seg.orientation,
      x: seg.cx,
      y: seg.cy,
    }));
}

/**
 * MVP:
 * - якщо waypoint ще нема -> створюємо 1 waypoint
 * - якщо waypoint є -> рухаємо перший waypoint
 *
 * Це простий старт.
 * Потім можна зробити точне зіставлення handle -> nearest waypoint.
 */
export function moveElbowSegmentHandle(connector, handle, pos) {
  const waypoints = [...(connector.route?.waypoints || [])];

  if (!waypoints.length) {
    waypoints.push({
      id: crypto.randomUUID(),
      x: handle.x,
      y: handle.y,
    });
  }

  const wp = waypoints[0];

  if (handle.orientation === "horizontal") {
    wp.y = pos.y;
  } else if (handle.orientation === "vertical") {
    wp.x = pos.x;
  }

  return {
    ...connector,
    route: {
      ...connector.route,
      waypoints,
    },
  };
}

export function createWaypointAtSegment(connector, segmentIndex) {
  const points = connector.geometry?.points || [];
  const i = segmentIndex * 2;

  if (i + 3 >= points.length) return null;

  const x1 = points[i];
  const y1 = points[i + 1];
  const x2 = points[i + 2];
  const y2 = points[i + 3];

  return {
    id: crypto.randomUUID(),
    x: (x1 + x2) / 2,
    y: (y1 + y2) / 2,
  };
}

        const isSelected =
          selection?.type === "edge" && selection?.id === connector.id;

        const startPoint = {
          x: pts[0],
          y: pts[1],

          y: pts[pts.length - 1],

          <Group key={connector.id}>

                e.cancelBubble = true;

# 7. Обмеження цього MVP

Тут важливо чесно сказати: це **перший робочий варіант**, а не фінальна модель.

### Поточне спрощення:

`moveElbowSegmentHandle()` зараз працює через **перший waypoint**.

Тобто:

* для однієї базової ломаної — ок
* для багатьох waypoint — ще неідеально

Але для MVP це правильно, бо:

* ти побачиш UX
* перевіриш, чи тобі заходить нова концепція
* не загрузнеш у складну математику одразу
